# Heimdall

```bash
mkdir -p /opt/docker/heimdall/config
```

```bash
nano /opt/docker/heimdall/docker-compose.yml
```

```bash
services:
  heimdall:
    image: lscr.io/linuxserver/heimdall:latest
    container_name: heimdall
    restart: unless-stopped

    ports:
      - "8080:80"

    environment:
      - PUID=1000
      - PGID=1000
      - TZ=America/Sao_Paulo

    volumes:
      - ./config:/config

    networks:
      - services

networks:
  services:
    external: true
```

```bash
cd /opt/docker/heimdall
```

```bash
docker compose up -d
```

# Home-Assistant

```bash
mkdir -p /opt/docker/homeassistant/config
```

```bash
nano /opt/docker/homeassistant/docker-compose.yml
```

```bash
services:
  homeassistant:
    image: ghcr.io/home-assistant/home-assistant:stable
    container_name: homeassistant
    restart: unless-stopped

    network_mode: host

    environment:
      TZ: America/Sao_Paulo

    volumes:
      - ./config:/config
      - /etc/localtime:/etc/localtime:ro

networks:
  services:
    external: true
```

```bash
cd /opt/docker/homeassistant
```

```bash
docker compose up -d
```
